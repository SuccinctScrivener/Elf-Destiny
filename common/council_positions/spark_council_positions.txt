councillor_aeluran_advisor = {
	skill = learning

	auto_fill = yes
	fill_from_pool = no
	inherit = no
	
	name = {
		desc = councillor_aeluran_advisor
	}

	tooltip = {
		desc = game_concept_aeluran_advisor_desc
	}

	valid_position = {
		# has_diarchy_active_parameter = vizier_sits_in_the_council_instead_of_spouse
	}

	valid_character = {
		exists = root.liege_or_court_owner
		# can_be_councillor_basics_trigger = yes
		save_temporary_scope_as = candidate

		is_aeluran_sister_or_higher = yes

		has_variable = appointed_to_advise
		var:appointed_to_advise = liege

		# trigger_if = {
		# 	limit = {
		# 		liege = {
		# 			has_realm_law_flag = aeluran_regency_enabled
		# 		}
		# 	}
		# 	liege = {
		# 		scope:candidate = diarch
		# 		has_diarchy_type = aeluran_regency
		# 	}
		# }
	}

	on_get_position = {
		root.liege = {
			set_variable = {
				name = religious_advisor
				value = root
			}

			add_character_flag = { # So ruler doesnt instantly get reinforcements
				flag = received_aeluran_magi_recently
				years = 2
			}
		}
	}

	on_lose_position = {
		root.liege = {
			remove_variable = religious_advisor
			if = {
				limit = {
					has_active_diarchy = yes
				}
				end_diarchy = yes
			}
		}
	}

	on_fired_from_position = {
		root.liege = {
			remove_variable = religious_advisor
			if = {
				limit = {
					has_active_diarchy = yes
				}
				end_diarchy = yes
			}
		}
	}

	barbershop_data = {
		position = { 763.0 261.00  }
		click_to_front = yes
	}
}
